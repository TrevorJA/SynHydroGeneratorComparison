# Python dependencies for the SynHydroGeneratorComparison pipeline.
#
# ---------------------------------------------------------------------
# How to use this file
# ---------------------------------------------------------------------
# This file lists DIRECT top-level dependencies only. It is intentionally
# unpinned so that it captures intent ("we need numpy / pandas / h5py /
# ...") independent of any particular cluster's build toolchain.
#
# For reproducibility on a specific cluster (e.g., Hopper), create a
# pinned lockfile from the environment that actually runs:
#
#   pip freeze --exclude-editable > requirements-lock.txt
#
# Check requirements-lock.txt into the repo for any publication run.
# The unpinned requirements.txt stays as the source of intent; the
# locked file captures the exact versions used.
#
# ---------------------------------------------------------------------
# Hopper environment setup (module load + conda env)
# ---------------------------------------------------------------------
# module purge
# module load anaconda3            # or whichever Python module Hopper
#                                  # recommends for the target compiler
# conda create -n synhydro python=3.11 -y
# conda activate synhydro
# pip install -r requirements.txt
# pip install -e /path/to/synhydro   # the internal library, editable
